    <script>
        const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbyBxEl71fkgNZ3QUMACs9soxRzMHy8KDWsgw7bgXAyLoRm7bb1mkUnhMkuW7CNLf5I0sw/exec";
        let allRanks = []; let allClasses = []; let globalUsers = []; let allGacha = [];

        window.onload = fetchData;
        function switchTab(id, el) { document.querySelectorAll('.tab-content').forEach(t=>t.classList.remove('active')); document.querySelectorAll('.tab').forEach(t=>t.classList.remove('active')); document.getElementById(id).classList.add('active'); el.classList.add('active'); }
        function toggleCostumeMenu() { const type = document.getElementById('sItemType').value; document.getElementById('costumeMenu').style.display = (type === 'costume' || type === 'avatar') ? "block" : "none"; }

        async function fetchData() {
            try {
                const data = await (await fetch(SCRIPT_URL)).json();
                globalUsers = data.users || [];
                let redeemHTML = ""; if(data.redemptions) { data.redemptions.reverse().forEach(r => { if(r.Status !== "รับของแล้ว") redeemHTML += `<tr><td>${r.Name}</td><td><b>${r.ItemName}</b></td><td><button onclick="approveRedeem('${r.RedeemID}')" class="btn" style="padding:5px;">รับของแล้ว</button></td></tr>`; }); } document.getElementById('redeemTable').innerHTML = redeemHTML;
                let userHTML = ""; globalUsers.forEach(u => { userHTML += `<tr><td>${u.UserID}</td><td>${u.ClassName||'-'}</td><td>${u.Avatar} ${u.DisplayName}</td><td><b>${u.TotalPoints}</b></td><td><b style="color:#ef6c00;">${u.Coins}</b></td></tr>`; }); document.getElementById('userTable').innerHTML = userHTML;
                let cpHTML = ""; if(data.coupons) { data.coupons.forEach(c => { cpHTML += `<tr><td><b>${c.Code}</b></td><td>+${c.CoinReward}</td><td>${c.CurrentUses} / ${c.MaxUses}</td></tr>`; }); } document.getElementById('couponTable').innerHTML = cpHTML;
                
                const b = data.settings.find(s=>s.SettingKey==='LevelBaseExp'); if(b) document.getElementById('lvlBase').value = b.SettingValue;
                const s = data.settings.find(s=>s.SettingKey==='LevelIncrement'); if(s) document.getElementById('lvlStep').value = s.SettingValue;
                const bn = data.settings.find(s=>s.SettingKey==='Level5Bonus'); if(bn) document.getElementById('lvlBonus').value = bn.SettingValue;

                allClasses = data.classes || []; renderClasses();
                allRanks = data.ranks || []; renderRanks();
                allGacha = data.gacha || []; renderGacha();
            } catch(e){}
        }

        async function sendAPI(payload, cb) { document.getElementById('loading').style.display = "block"; await fetch(SCRIPT_URL, { method: 'POST', mode: 'no-cors', headers: {'Content-Type':'application/json'}, body: JSON.stringify(payload) }); document.getElementById('loading').style.display = "none"; cb(); }
        function approveRedeem(id) { if(confirm("ยืนยันให้รางวัล?")) sendAPI({ action: "updateRedeem", redeemId: id, status: "รับของแล้ว" }, () => { fetchData(); }); }
        
        function addQuest() { sendAPI({ action: "addQuest", questId: "Q"+Date.now(), title: document.getElementById('qTitle').value, type: document.getElementById('qType').value, content: document.getElementById('qContent').value, difficulty: document.getElementById('qDiff').value, points: document.getElementById('qExp').value + "," + document.getElementById('qCoins').value , isSet: "false", assignedClasses: "all", itemRewardID: document.getElementById('qItemRewardID').value }, () => { alert("บันทึกสำเร็จ"); fetchData(); }); }
        function addCoupon() { sendAPI({ action: "addCoupon", couponId: "CP"+Date.now(), code: document.getElementById('cpCode').value, coinReward: document.getElementById('cpReward').value, maxUses: document.getElementById('cpMaxUses').value }, () => { alert("สร้างคูปองสำเร็จ!"); fetchData(); }); }

        function addShopItem() {
            let itemIdInput = document.getElementById('sItemId').value.trim();
            let finalItemId = itemIdInput ? itemIdInput : "S" + Date.now(); // ถ้าไม่ใส่รหัส ให้สุ่มอัตโนมัติ
            
            let itemType = document.getElementById('sItemType').value; 
            let layerType = (itemType === 'costume') ? document.getElementById('sLayerType').value : (itemType === 'avatar' ? 'partner' : 'none');
            // ตัวอย่างการเพิ่มค่า Limit ส่งไปให้ระบบหลังบ้าน
        let payload = {
            action: "addShopItem", // หรือชื่อ action ที่คุณครูใช้อยู่
            /* ... ข้อมูลอื่นๆ เช่น ชื่อ ราคา ... */
            limitPerMonth: document.getElementById('iLimit').value // 🌟 เพิ่มบรรทัดนี้
        };
            sendAPI({ action: "addShopItem", itemId: finalItemId, name: document.getElementById('sName').value, points: document.getElementById('sPoints').value, icon: document.getElementById('sIcon').value||"🎁", itemType: itemType, layerType: layerType, layerImage: document.getElementById('sLayerImage').value, compatibleChars: document.getElementById('sCompatibleChars').value }, () => { alert("บันทึกสินค้าลงร้านสำเร็จ"); fetchData(); });
        }

        function addGachaRow() { allGacha.push({BoxLevel:"", ItemID:"", DropRatePercent:""}); renderGacha(); }
        function saveGachaConfig() {
            let boxes = document.querySelectorAll('.g-box'); let items = document.querySelectorAll('.g-item'); let rates = document.querySelectorAll('.g-rate'); let newGacha = [];
            for(let i=0; i<boxes.length; i++) { if(items[i].value.trim() && boxes[i].value.trim()) newGacha.push({boxLevel: boxes[i].value.trim(), itemId: items[i].value.trim(), dropRate: rates[i].value}); }
            sendAPI({ action: "saveGachaConfig", gachaData: newGacha }, () => { alert("บันทึกตารางกล่องสุ่มสำเร็จ!"); fetchData(); });
